async function loadComments() { const response = await fetch(sheetURL); const data = await response.text(); // Regex ini memecah baris berdasarkan baris baru (\n atau \r\n) // TAPI mengabaikan baris baru yang berada di dalam tanda kutip dua (isi sel). const rows = data.match(/(".*?"|[^",\s]+)(?=\s*,|\s*$)/gm) || []; /* Pendekatan Alternatif yang lebih sederhana namun akurat: Memecah string berdasarkan baris baru yang tidak berada di dalam kutipan. */ const cleanRows = data.trim().split(/\r?\n(?=(?:(?:[^"]*"){2})*[^"]*$)/).slice(1); let html = ''; cleanRows.reverse().forEach(row => { // Mengambil isi Kolom A dengan menangani kasus teks yang dibungkus tanda kutip let columnAContent = ''; // Regex untuk mengambil elemen pertama sebelum koma pertama yang tidak di dalam kutipan const match = row.match(/^(".*?"|[^,]+)/); if (match) { columnAContent = match[0] .replace(/^"|"$/g, '') // Menghapus kutipan di awal/akhir .replace(/""/g, '"') // Mengubah double-quote CSV menjadi single-quote .trim(); } if (columnAContent) { html += `
${columnAContent.replace(/\n/g, '
')}
`; } }); document.getElementById('comments').innerHTML = html; }
