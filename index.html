async function loadComments() { try { const response = await fetch(sheetURL); const data = await response.text(); // Menggunakan regex untuk memecah baris agar sel yang memiliki // "Enter" di dalamnya tidak terpecah menjadi blok baru. const rows = data.trim().split(/\r?\n(?=(?:(?:[^"]*"){2})*[^"]*$)/).slice(1); let html = ''; rows.reverse().forEach(row => { // Mengambil hanya kolom A // Regex ini menangkap teks di dalam kutipan atau teks sebelum koma pertama const match = row.match(/^(".*?"|[^,]+)/); let columnAContent = ''; if (match) { columnAContent = match[0] .replace(/^"|"$/g, '') // Hapus kutipan pembungkus .replace(/""/g, '"') // Ubah double quote menjadi single .trim(); } if (columnAContent) { html += `
${columnAContent.replace(/\n/g, '
')}
`; } }); document.getElementById('comments').innerHTML = html; } catch (error) { console.error("Gagal memuat data:", error); document.getElementById('comments').innerHTML = "Gagal memuat referensi data."; } }
