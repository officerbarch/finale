async function loadComments() { const response = await fetch(sheetURL); const data = await response.text(); // Memecah berdasarkan baris (setiap baris merepresentasikan satu baris di Spreadsheet) const rows = data.trim().split('\n').slice(1); let html = ''; rows.reverse().forEach(row => { // Logika untuk mengambil HANYA kolom A: // Kita mencari indeks koma pertama untuk memisahkan Kolom A dengan kolom lainnnya const firstComma = row.indexOf(','); let columnAContent = ''; if (firstComma !== -1) { // Mengambil teks sebelum koma pertama (Kolom A) columnAContent = row.substring(0, firstComma).replace(/^"|"$/g, '').trim(); } else { // Jika tidak ada koma, berarti hanya ada satu kolom columnAContent = row.replace(/^"|"$/g, '').trim(); } // Membuat 1 box untuk 1 sel di kolom A if (columnAContent) { html += `
${columnAContent}
`; } }); document.getElementById('comments').innerHTML = html; }
